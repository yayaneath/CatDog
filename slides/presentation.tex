	\section{\scshape Experimentación}
	
	\begin{frame}
		\frametitle{Configuración}
	
		Entrenamiento utilizando 25.000 imágenes:
	
		\begin{itemize}
			\item 90\% reservado para entrenamiento de la red, restante para validación
			
			\item 30 iteraciones
			
			\item Imágenes transformadas a 150 x 150 x 3
			
			\item 44.497.985 de parámetros o pesos a aprender por la topología de la \textit{CNN}
			
			\item Ejecutado sobre GTX 960 4GB (1024 CUDA cores)
			
			\item Aproximadamente 75 minutos en ejecución
			
			\item 86\% acierto sobre conjunto de validación
		\end{itemize}
	\end{frame}
